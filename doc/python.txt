################################### conda ####################################
# create virtual env
conda create -n ENV [python=x.x] anaconda
# list env
conda info -e
# activate it
source activate ENV
# install packages
conda install -n ENV <package>
# deactivate
conda deactivate
# delete virtual env
conda remove -n ENV -all


conda update conda
